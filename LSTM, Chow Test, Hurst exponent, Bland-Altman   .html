    X, y = [], []

        y.append(data[i+time_steps])

#This is to run the Chow Test for Structural Breaks
import statsmodels.api as sm

# Create a time variable
df['time'] = range(1, len(df) + 1)

# Split the data into two segments for the structural break test
split_point = len(df) // 2

# Define the dependent and independent variables for each segment
y1 = df['rain'][:split_point]
X1 = sm.add_constant(df['time'][:split_point])
y2 = df['rain'][split_point:]
X2 = sm.add_constant(df['time'][split_point:])

# Fit linear regressions for each segment
model1 = sm.OLS(y1, X1).fit()
model2 = sm.OLS(y2, X2).fit()

# Perform the Chow test for structural break
chow_test = sm.stats.anova_lm(model1, model2)

# Display the Chow test results
print("Chow Test Statistic:", chow_test['F'][0])
print("p-value:", chow_test['Pr(>F)'][0])

# Determine if a structural break is detected based on the p-value
alpha = 0.05  # Significance level
if chow_test['Pr(>F)'][0] < alpha:
    print("A structural break is detected (reject null hypothesis)")
else:
    print("No structural break is detected (fail to reject null hypothesis)")

#This is to run the R/S analysis. The Hurst exponent test for non-linearity
#I am running the Hurst exponent test because the result from the Chow test failed
# to reject the null hypothesis of no structural breaks

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load the time series data (replace 'your_data' with your actual data)
data = df['rain'].values  # Make sure 'rain' matches the column name in your dataset

# Define a function to calculate the Hurst exponent
def hurst_exponent(data, max_lag=20):
    lags = range(2, max_lag)
    tau = [np.std(np.subtract(data[lag:], data[:-lag])) for lag in lags]
    hurst = np.polyfit(np.log(lags), np.log(tau), 1)[0]

    return hurst

# Calculate the Hurst exponent
hurst_value = hurst_exponent(data)

print("Hurst Exponent:", hurst_value)

# Interpret the results
if hurst_value < 0.5:
    print("Hurst Exponent < 0.5: The time series is mean-reverting (trending)")
elif hurst_value == 0.5:
    print("Hurst Exponent = 0.5: The time series follows a Brownian motion (random walk)")
else:
    print("Hurst Exponent > 0.5: The time series is a persistent (trending) time series")

# Plot a log-log graph for R/S analysis (optional)
def plot_rs_analysis(data):
    lags = range(2, 20)
    tau = [np.std(np.subtract(data[lag:], data[:-lag])) for lag in lags]

    plt.figure()
    plt.plot(np.log(lags), np.log(tau), marker='o')
    plt.xlabel('log(Lag)')
    plt.ylabel('log(R/S)')
    plt.title('R/S Analysis')
    plt.show()

plot_rs_analysis(data)

#the Null Hypothesis is 0 breaks
#the Alternative Hypothesis is at least 1 break

import statsmodels.api as sm

# Create a time variable
df['time'] = range(1, len(df) + 1)

# Define the split point (adjust this as needed)
split_point = len(df) // 2

# Check if the split point is within a valid range
if split_point < 1 or split_point >= len(df):
    print("Invalid split point. Adjust the split_point variable.")
else:
    # Define the dependent and independent variables for each segment
    y1 = df['rain'][:split_point]
    X1 = sm.add_constant(df['time'][:split_point])
    y2 = df['rain'][split_point:]
    X2 = sm.add_constant(df['time'][split_point:])

    # Fit linear regressions for each segment
    model1 = sm.OLS(y1, X1).fit()
    model2 = sm.OLS(y2, X2).fit()

    # Perform the Chow test for structural break
    chow_test = sm.stats.anova_lm(model1, model2)

    # Display the Chow test results
    print("Chow Test Statistic:", chow_test['F'][0])
    print("p-value:", chow_test['Pr(>F)'][0])

    # Determine if a structural break is detected based on the p-value
    alpha = 0.05  # Significance level
    if chow_test['Pr(>F)'][0] < alpha:
        print("A structural break is detected (reject null hypothesis)")
    else:
        print("No structural break is detected (fail to reject null hypothesis)")

upper_limit = mean_difference + 1.96 * differences.std()

# Create the Bland-Altman plot
plt.figure(figsize=(10, 6))

plt.axhline(y=mean_difference, color='red', linestyle='--', label='Mean Difference (Bias)')
