fnv = df['fnv'].values  # For what data you have replace 'fnv' with 'your variable'

# Data preprocessing
scaler = MinMaxScaler()
fnv = scaler.fit_transform(fnv.reshape(-1, 1))  # Ensure the scaler operates on 'fnv'

# Define a function to prepare the data for LSTM
def prepare_data(data, time_steps):
    X, y = [], []
    for i in range(len(data) - time_steps):
        X.append(data[i:i+time_steps])
        y.append(data[i+time_steps])
    return np.array(X), np.array(y)

# Set the number of time steps and split the data into train and test sets
time_steps = 10  # Adjust as needed
X, y = prepare_data(fnv, time_steps)  # Use 'fnv' instead of 'WTI'
train_size = int(0.8 * len(X))
X_train, X_test = X[:train_size], X[train_size:]
y_train, y_test = y[:train_size], y[train_size:]

# Build the LSTM model
model = Sequential()
model.add(LSTM(units=50, activation='relu', input_shape=(X_train.shape[1], 1)))
model.add(Dense(units=1))
model.compile(optimizer=Adam(learning_rate=0.001), loss='mean_squared_error')

# Train the model
model.fit(X_train, y_train, epochs=50, batch_size=32, verbose=1)

# Evaluate the model on the test data
test_loss = model.evaluate(X_test, y_test)
print(f'Test Loss: {test_loss}')

# Forecast future values
future_time_steps = 12  # Adjust as needed
forecast = []

# Use the last part of the training data to start forecasting
current_sequence = X_train[-1]

for i in range(future_time_steps):
    next_value = model.predict(current_sequence.reshape(1, time_steps, 1))
    forecast.append(next_value[0, 0])
    current_sequence = np.append(current_sequence[1:], next_value)

# Inverse transform the forecasted values to the original scale
forecast = scaler.inverse_transform(np.array(forecast).reshape(-1, 1))

# Plot the forecasted values
plt.figure(figsize=(12, 6))
plt.plot(df.index[-future_time_steps:], forecast, label='Forecast', color='red')
#plt.plot(df.index[-future_time_steps:], df['RER'][-future_time_steps:], label='Actual', color='blue')
plt.xlabel('Date')
plt.ylabel('Food and Vegetables Imports')  # Update the ylabel to 'Food and Vegetables Imports'
plt.legend()
plt.show()

# Create a DataFrame to store the forecasts
forecast_df = pd.DataFrame()
forecast_df['Date'] = df.index[-future_time_steps:]
forecast_df['Forecast'] = forecast

# Display the DataFrame
print(forecast_df)

#this is for the Bland-Altman Plot
import matplotlib.pyplot as plt

# Calculate the differences between predicted and actual values
differences = y_test_original - test_predictions

# Calculate the mean difference (bias)
mean_difference = differences.mean()

# Calculate the limits of agreement (LoA)
lower_limit = mean_difference - 1.96 * differences.std()
upper_limit = mean_difference + 1.96 * differences.std()

# Create the Bland-Altman plot
plt.figure(figsize=(10, 6))
plt.scatter(y_test_original, differences, color='blue', alpha=0.7)
plt.axhline(y=mean_difference, color='red', linestyle='--', label='Mean Difference (Bias)')
plt.axhline(y=lower_limit, color='green', linestyle='--', label='Lower Limit of Agreement')
plt.axhline(y=upper_limit, color='green', linestyle='--', label='Upper Limit of Agreement')
plt.xlabel('Actual Values')
plt.ylabel('Differences (Actual - Predicted)')
plt.title('Bland-Altman Plot')
plt.legend()
plt.grid(True)
plt.show()
