import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.preprocessing import MinMaxScaler
from sklearn.metrics import mean_squared_error
import math

from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout
from tensorflow.keras.optimizers import Adam
from tensorflow.keras.callbacks import EarlyStopping

# === Load the data ===
df = pd.read_excel(r"G:\2018\Documents\papers\Portfolio finance\LSTM\Ammonia.xlsx")

# === Preprocess the data ===
ammonia = df['Ammonia'].values  # Replace with your actual DataFrame and column
scaler = MinMaxScaler()
ammonia_scaled = scaler.fit_transform(ammonia.reshape(-1, 1))

# === Prepare data for LSTM ===
def prepare_data(data, time_steps):
    X, y = [], []
    for i in range(len(data) - time_steps):
        X.append(data[i:i+time_steps])
        y.append(data[i+time_steps])
    return np.array(X), np.array(y)

def train_test_split_series(data, time_steps, split_ratio=0.8):
    X, y = prepare_data(data, time_steps)
    train_size = int(split_ratio * len(X))
    return X[:train_size], X[train_size:], y[:train_size], y[train_size:]

# === Set time steps and split data ===
time_steps = 10
X_train, X_test, y_train, y_test = train_test_split_series(ammonia_scaled, time_steps)

# Reshape for LSTM [samples, time steps, features]
X_train = X_train.reshape((X_train.shape[0], X_train.shape[1], 1))
X_test = X_test.reshape((X_test.shape[0], X_test.shape[1], 1))

# === Build LSTM model ===
model = Sequential()
model.add(LSTM(units=50, activation='relu', input_shape=(time_steps, 1)))
model.add(Dropout(0.2))
model.add(Dense(units=1))
model.compile(optimizer=Adam(learning_rate=0.001), loss='mean_squared_error')

# === Train the model with EarlyStopping ===
early_stop = EarlyStopping(monitor='val_loss', patience=5, restore_best_weights=True)

model.fit(X_train, y_train,
          epochs=100,
          batch_size=32,
          validation_split=0.1,
          callbacks=[early_stop],
          verbose=1)

# === Evaluate the model ===
test_loss = model.evaluate(X_test, y_test, verbose=0)
print(f'Test Loss (MSE): {test_loss}')

# Calculate RMSE
y_pred = model.predict(X_test)
rmse = math.sqrt(mean_squared_error(y_test, y_pred))
print(f'RMSE: {rmse}')

# === Forecast future values ===
future_time_steps = 12
forecast = []

current_sequence = X_train[-1].copy()

for _ in range(future_time_steps):
    prediction = model.predict(current_sequence.reshape(1, time_steps, 1), verbose=0)
    forecast.append(prediction[0, 0])
    current_sequence = np.roll(current_sequence, -1)
    current_sequence[-1] = prediction

# Inverse transform to original scale
forecast = scaler.inverse_transform(np.array(forecast).reshape(-1, 1))

# === Generate forecast dates starting from March 2025 ===
forecast_start_date = pd.to_datetime('2025-03-01')
forecast_dates = pd.date_range(start=forecast_start_date, periods=future_time_steps, freq='MS')

# === Plot forecast ===
plt.figure(figsize=(12, 6))
plt.plot(forecast_dates, forecast, label='Forecast', color='red')
plt.title('Ammonia Price Forecast')
plt.xlabel('Date')
plt.ylabel('Ammonia Prices')
plt.grid(True)
plt.legend()
plt.show()

# === Store forecast in DataFrame ===
forecast_df = pd.DataFrame({
    'Date': forecast_dates,
    'Forecast': forecast.flatten()
})

# === Display forecast ===
print(forecast_df)

#This is to run the Chow Test for Structural Breaks
import statsmodels.api as sm

# Create a time variable
df['Date'] = range(1, len(df) + 1)

# Split the data into two segments for the structural break test
split_point = len(df) // 2

# Define the dependent and independent variables for each segment
y1 = df['Ammonia'][:split_point]
X1 = sm.add_constant(df['Date'][:split_point])
y2 = df['Ammonia'][split_point:]
X2 = sm.add_constant(df['Date'][split_point:])

# Fit linear regressions for each segment
model1 = sm.OLS(y1, X1).fit()
model2 = sm.OLS(y2, X2).fit()

# Perform the Chow test for structural break
chow_test = sm.stats.anova_lm(model1, model2)

# Display the Chow test results
print("Chow Test Statistic:", chow_test['F'][0])
print("p-value:", chow_test['Pr(>F)'][0])

# Determine if a structural break is detected based on the p-value
alpha = 0.05  # Significance level
if chow_test['Pr(>F)'][0] < alpha:
    print("A structural break is detected (reject null hypothesis)")
else:
    print("No structural break is detected (fail to reject null hypothesis)")

#This is to run the R/S analysis. The Hurst exponent test for non-linearity
#I am running the Hurst exponent test because the result from the Chow test failed
# to reject the null hypothesis of no structural breaks

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load the time series data (replace 'your_data' with your actual data)
data = df['Ammonia'].values  # Make sure 'rain' matches the column name in your dataset

# Define a function to calculate the Hurst exponent
def hurst_exponent(data, max_lag=20):
    lags = range(2, max_lag)
    tau = [np.std(np.subtract(data[lag:], data[:-lag])) for lag in lags]
    hurst = np.polyfit(np.log(lags), np.log(tau), 1)[0]

    return hurst

# Calculate the Hurst exponent
hurst_value = hurst_exponent(data)

print("Hurst Exponent:", hurst_value)

# Interpret the results
if hurst_value < 0.5:
    print("Hurst Exponent < 0.5: The time series is mean-reverting (trending)")
elif hurst_value == 0.5:
    print("Hurst Exponent = 0.5: The time series follows a Brownian motion (random walk)")
else:
    print("Hurst Exponent > 0.5: The time series is a persistent (trending) time series")

# Plot a log-log graph for R/S analysis (optional)
def plot_rs_analysis(data):
    lags = range(2, 20)
    tau = [np.std(np.subtract(data[lag:], data[:-lag])) for lag in lags]

    plt.figure()
    plt.plot(np.log(lags), np.log(tau), marker='o')
    plt.xlabel('log(Lag)')
    plt.ylabel('log(R/S)')
    plt.title('R/S Analysis')
    plt.show()

plot_rs_analysis(data)

#this is for the Bland-Altman Plot
import matplotlib.pyplot as plt

# Calculate the differences between predicted and actual values
differences = y_test_original - test_predictions

# Calculate the mean difference (bias)
mean_difference = differences.mean()

# Calculate the limits of agreement (LoA)
lower_limit = mean_difference - 1.96 * differences.std()
upper_limit = mean_difference + 1.96 * differences.std()

# Create the Bland-Altman plot
plt.figure(figsize=(10, 6))
plt.scatter(y_test_original, differences, color='blue', alpha=0.7)
plt.axhline(y=mean_difference, color='red', linestyle='--', label='Mean Difference (Bias)')
plt.axhline(y=lower_limit, color='green', linestyle='--', label='Lower Limit of Agreement')
plt.axhline(y=upper_limit, color='green', linestyle='--', label='Upper Limit of Agreement')
plt.xlabel('Actual Values')
plt.ylabel('Differences (Actual - Predicted)')
plt.title('Bland-Altman Plot')
plt.legend()
plt.grid(True)
